create table ACTIONS
(
	ID INT	            NOT NULL,
	ACTION INT          NOT NULL,
	FILE VARCHAR(128)	NOT NULL,
	CREATED	DATETIME	NOT NULL,
	PRIMARY KEY (ID)
);

create table FILES
(
	ID INT				  NOT NULL,
	NAME VARCHAR(128)	  NOT NULL,
	KEY  CHAR(128)	      NOT NULL,
	LASTUPDATE DATETIME   NOT NULL,
	PRIMARY KEY(ID)
);

CREATE INDEX KEY_INDEX ON FILES (KEY);

CREATE INDEX NAME_INDEX ON FILES (NAME);
